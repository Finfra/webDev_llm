# 2.6. 모듈화와 프로젝트 구조화

## CommonJS/ESM 차이
* CommonJS (Node.js 기본 모듈 시스템)
    - `require()`로 모듈 가져오기
    - `module.exports`로 모듈 내보내기
    - 동기적 로딩 방식
    - Node.js에서 기본적으로 사용
| 항목        | CommonJS              | ESM (ECMAScript Module)                  |
| ----------- | --------------------- | ---------------------------------------- |
| 도입 시기   | Node.js 초기부터      | ES6 (2015) 표준부터                      |
| 주 사용처   | Node.js (기본)        | 브라우저, Node.js (지원 강화됨)          |
| 파일 확장자 | `.js` (기본)          | `.mjs`, 또는 `type: module` 설정된 `.js` |
| 모듈 형식   | 동기 방식 (`require`) | 비동기 방식 (`import`)                   |

    - 
```javascript
// 내보내기 (math.js)
// 파일명: 2.6.src/1.js
const add = (a, b) => a + b;
const subtract = (a, b) => a - b;

module.exports = {
  add,
  subtract
};

// 또는 개별적으로 내보내기
module.exports.multiply = (a, b) => a * b;

// 가져오기 (app.js)
const math = require('./math');
console.log(math.add(5, 3)); // 8

// 구조 분해 할당으로 가져오기
const { subtract, multiply } = require('./math');
console.log(subtract(10, 4)); // 6
console.log(multiply(2, 3)); // 6
```

* ESM (ECMAScript Modules)
    - `import`로 모듈 가져오기
    - `export`로 모듈 내보내기
    - 비동기적 로딩 가능
    - 브라우저 호환, 정적 분석 가능
    - Node.js에서는 `.mjs` 확장자 사용 또는 package.json에 `"type": "module"` 설정 필요

```javascript
// 내보내기 (math.mjs)
// 파일명: 2.6.src/1.mjs
export const add = (a, b) => a + b;
export const subtract = (a, b) => a - b;

// 기본 내보내기
export default function multiply(a, b) {
  return a * b;
}

// 가져오기 (app.mjs)
// 파일명: 2.6.src/2.mjs
import { add, subtract } from './math.mjs';
import multiply from './math.mjs'; // 기본 내보내기 가져오기

console.log(add(5, 3)); // 8
console.log(subtract(10, 4)); // 6
console.log(multiply(4, 2)); // 8
```

* CommonJS와 ESM 비교

| 특성        | CommonJS                | ESM                         |
| ----------- | ----------------------- | --------------------------- |
| 문법        | require, module.exports | import, export              |
| 로딩 방식   | 동기                    | 비동기 (정적 가져오기)      |
| 사용 환경   | Node.js                 | 브라우저, Node.js           |
| 코드 분석   | 런타임                  | 정적 (빌드타임)             |
| 파일 확장자 | .js                     | .mjs 또는 .js(type: module) |
| 조건부 로딩 | 가능                    | 최상위 레벨에서 불가능      |

* package.json에서 ESM 활성화
```json
{
  "name": "my-project",
  "version": "1.0.0",
  "type": "module",
  "main": "index.js"
}
```

## 디렉토리 나누기, 모듈 export/import

* 권장되는 프로젝트 구조
```
my-project/
  |- package.json
  |- index.js         # 애플리케이션 진입점
  |- config/          # 설정 파일
  |    |- config.js
  |- models/          # 데이터 모델
  |    |- user.js
  |    |- product.js
  |- controllers/     # 비즈니스 로직
  |    |- userController.js
  |    |- productController.js
  |- routes/          # 라우팅 로직
  |    |- userRoutes.js
  |    |- productRoutes.js
  |- utils/           # 유틸리티 함수
  |    |- helpers.js
  |    |- logger.js
  |- middlewares/     # 미들웨어
  |    |- auth.js
  |- tests/           # 테스트 코드
  |    |- user.test.js
  |- public/          # 정적 파일
  |    |- css/
  |    |- js/
  |    |- images/
  |- views/           # 템플릿 파일
       |- index.html
```

* CommonJS 모듈 예제

```javascript
// ./config/config.js
module.exports = {
  port: process.env.PORT || 3000,
  dbUrl: process.env.DB_URL || 'mongodb://localhost:27017/myapp',
  jwtSecret: process.env.JWT_SECRET || 'your-secret-key'
};

// ./utils/logger.js
function logInfo(message) {
  console.log(`[INFO] ${new Date().toISOString()}: ${message}`);
}

function logError(message) {
  console.error(`[ERROR] ${new Date().toISOString()}: ${message}`);
}

module.exports = {
  logInfo,
  logError
};

// ./models/user.js
const users = [];

function findUserById(id) {
  return users.find(user => user.id === id);
}

function createUser(userData) {
  const newUser = {
    id: Date.now().toString(),
    ...userData
  };
  users.push(newUser);
  return newUser;
}

module.exports = {
  findUserById,
  createUser
};

// ./controllers/userController.js
const { findUserById, createUser } = require('../models/user');
const { logInfo, logError } = require('../utils/logger');

function getUser(req, res) {
  const userId = req.params.id;
  const user = findUserById(userId);
  
  if (user) {
    logInfo(`User found: ${userId}`);
    return user;
  } else {
    logError(`User not found: ${userId}`);
    return null;
  }
}

function addUser(userData) {
  try {
    const newUser = createUser(userData);
    logInfo(`User created: ${newUser.id}`);
    return newUser;
  } catch (error) {
    logError(`Error creating user: ${error.message}`);
    throw error;
  }
}

module.exports = {
  getUser,
  addUser
};

// index.js (애플리케이션 진입점)
const config = require('./config/config');
const { logInfo } = require('./utils/logger');
const userController = require('./controllers/userController');

logInfo(`Starting application on port ${config.port}`);

// 사용자 추가 테스트
const newUser = userController.addUser({ name: 'John Doe', email: 'john@example.com' });
console.log('새 사용자:', newUser);

// 사용자 조회 테스트 (가상의 요청/응답 객체)
const req = { params: { id: newUser.id } };
const user = userController.getUser(req);
console.log('조회된 사용자:', user);
```

* utils/fileUtils.js
```javascript
const fs = require('fs');
const path = require('path');
const config = require('../config/config');

// 디렉토리가 없으면 생성
function ensureDirectoryExists(filePath) {
  const dirname = path.dirname(filePath);
  if (!fs.existsSync(dirname)) {
    fs.mkdirSync(dirname, { recursive: true });
  }
}

// 할일 목록 로드
function loadData() {
  ensureDirectoryExists(config.dataFile);
  
  try {
    if (fs.existsSync(config.dataFile)) {
      const data = fs.readFileSync(config.dataFile, 'utf8');
      return JSON.parse(data);
    }
    return [];
  } catch (error) {
    console.error('파일 로드 중 오류:', error.message);
    return [];
  }
}

// 할일 목록 저장
function saveData(data) {
  ensureDirectoryExists(config.dataFile);
  
  try {
    fs.writeFileSync(config.dataFile, JSON.stringify(data, null, 2), 'utf8');
    return true;
  } catch (error) {
    console.error('파일 저장 중 오류:', error.message);
    return false;
  }
}

module.exports = {
  loadData,
  saveData
};
```

* utils/display.js
```javascript
const config = require('../config/config');

// 애플리케이션 헤더 표시
function showHeader() {
  console.log(`\n===== ${config.appName} =====`);
}

// 메인 메뉴 표시
function showMainMenu() {
  showHeader();
  console.log('1. 할일 목록 보기');
  console.log('2. 할일 추가하기');
  console.log('3. 할일 완료하기');
  console.log('4. 할일 삭제하기');
  console.log('0. 종료하기');
  console.log('=======================\n');
}

// 할일 목록 표시
function showTodoList(todos) {
  showHeader();
  if (todos.length === 0) {
    console.log('할일이 없습니다.');
  } else {
    todos.forEach((todo, index) => {
      const status = todo.completed ? '[완료]' : '[진행중]';
      console.log(`${index + 1}. ${status} ${todo.text}`);
    });
  }
  console.log('=====================\n');
}

module.exports = {
  showHeader,
  showMainMenu,
  showTodoList
};
```

* models/todo.js
```javascript
const { loadData, saveData } = require('../utils/fileUtils');

// 할일 조회
function getAllTodos() {
  return loadData();
}

// 할일 추가
function addTodo(text) {
  const todos = loadData();
  const newTodo = {
    id: Date.now().toString(),
    text,
    completed: false,
    createdAt: new Date().toISOString()
  };
  
  todos.push(newTodo);
  const success = saveData(todos);
  
  return success ? newTodo : null;
}

// 할일 완료 처리
function completeTodo(index) {
  const todos = loadData();
  
  if (index < 0 || index >= todos.length) {
    return null;
  }
  
  todos[index].completed = true;
  const success = saveData(todos);
  
  return success ? todos[index] : null;
}

// 할일 삭제
function deleteTodo(index) {
  const todos = loadData();
  
  if (index < 0 || index >= todos.length) {
    return null;
  }
  
  const deleted = todos.splice(index, 1)[0];
  const success = saveData(todos);
  
  return success ? deleted : null;
}

module.exports = {
  getAllTodos,
  addTodo,
  completeTodo,
  deleteTodo
};
```

* controllers/todoController.js
```javascript
const todoModel = require('../models/todo');
const display = require('../utils/display');
const readline = require('readline');

// readline 인터페이스 생성
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

// 애플리케이션 시작
function startApp() {
  display.showMainMenu();
  promptUser();
}

// 사용자 입력 처리
function promptUser() {
  rl.question('원하는 작업을 선택하세요: ', (choice) => {
    handleUserChoice(choice);
  });
}

// 사용자 선택 처리
function handleUserChoice(choice) {
  switch (choice) {
    case '0':
      console.log('프로그램을 종료합니다.');
      rl.close();
      break;
    case '1':
      showTodos();
      break;
    case '2':
      addTodo();
      break;
    case '3':
      completeTodo();
      break;
    case '4':
      deleteTodo();
      break;
    default:
      console.log('올바른 선택이 아닙니다. 다시 시도하세요.');
      display.showMainMenu();
      promptUser();
      break;
  }
}

// 할일 목록 표시
function showTodos() {
  const todos = todoModel.getAllTodos();
  display.showTodoList(todos);
  display.showMainMenu();
  promptUser();
}

// 할일 추가
function addTodo() {
  rl.question('추가할 할일을 입력하세요: ', (text) => {
    const newTodo = todoModel.addTodo(text);
    
    if (newTodo) {
      console.log(`'${text}' 항목이 추가되었습니다.`);
    } else {
      console.log('할일 추가 중 오류가 발생했습니다.');
    }
    
    display.showMainMenu();
    promptUser();
  });
}

// 할일 완료 처리
function completeTodo() {
  const todos = todoModel.getAllTodos();
  display.showTodoList(todos);
  
  if (todos.length === 0) {
    display.showMainMenu();
    promptUser();
    return;
  }
  
  rl.question('완료할 항목 번호를 입력하세요: ', (num) => {
    const index = parseInt(num) - 1;
    const completed = todoModel.completeTodo(index);
    
    if (completed) {
      console.log(`'${completed.text}' 항목이 완료 처리되었습니다.`);
    } else {
      console.log('유효하지 않은 항목 번호이거나 처리 중 오류가 발생했습니다.');
    }
    
    display.showMainMenu();
    promptUser();
  });
}

// 할일 삭제
function deleteTodo() {
  const todos = todoModel.getAllTodos();
  display.showTodoList(todos);
  
  if (todos.length === 0) {
    display.showMainMenu();
    promptUser();
    return;
  }
  
  rl.question('삭제할 항목 번호를 입력하세요: ', (num) => {
    const index = parseInt(num) - 1;
    const deleted = todoModel.deleteTodo(index);
    
    if (deleted) {
      console.log(`'${deleted.text}' 항목이 삭제되었습니다.`);
    } else {
      console.log('유효하지 않은 항목 번호이거나 처리 중 오류가 발생했습니다.');
    }
    
    display.showMainMenu();
    promptUser();
  });
}

module.exports = {
  startApp
};
```
